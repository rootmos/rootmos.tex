\DeclareFieldFormat{prenote}{\IfBlankTF{#1}{}{\protect\prenote{#1}}}
\DeclareFieldFormat{postnote}{\IfBlankTF{#1}{}{\protect\postnote{#1}}}

\NewDocumentCommand{\pc}{o o m}{%
    \IfNoValueTF{#1}{%
        \parencite{#3}%
    }{%
        \IfNoValueTF{#2}{%
            \parencite[][#1]{#3}%
        }{%
            \parencite[#1][#2]{#3}%
        }%
    }%
}

\NewDocumentCommand{\tc}{o o m}{%
    \IfNoValueTF{#1}{%
        \textcite{#3}%
    }{%
        \IfNoValueTF{#2}{%
            \textcite[][#1]{#3}%
        }{%
            \textcite[#1][#2]{#3}%
        }%
    }%
}

\NewDocumentCommand{\bq}{o o m +m}{%
    \blockcquote[#1][#2]{#3}{\ignorespaces#4\unskip}%
}

\NewDocumentCommand{\tq}{o o m +m}{%
    \textcquote[#1][#2]{#3}{\ignorespaces#4\unskip}%
}

\NewDocumentCommand{\pcin}{O{} o m O{} m}{%
    \mkbibparens{%
        \IfNoValueTF{#2}{%
            \nptextcite[#1]{#3}%
        }{%
            \nptextcite[#1][#2]{#3}%
        }%
        \addcomma\addspace\bibstring{quotedin}\addspace%
        \nptextcite[#4]{#5}%
    }%
}

\NewDocumentCommand{\tcin}{O{} o m O{} m}{%
    \citeauthor{#3}\addspace\mkbibparens{%
        \IfNoValueTF{#2}{%
            \citeyear[#1]{#3}%
        }{%
            \citeyear[#1][#2]{#3}%
        }%
        \addcomma\addspace\bibstring{quotedin}\addspace%
        \nptextcite[#4]{#5}%
    }%
}

\NewDocumentCommand{\bqin}{O{} o m O{} o m +m}{
    \blockquote{\ignorespaces#7\unskip%
        \addspace%
        \mkbibparens{%
            \IfNoValueTF{#2}{%
                \nptextcite[#1]{#3}%
            }{%
                \nptextcite[#1][#2]{#3}%
            }%
            \addcomma\addspace\bibstring{quotedin}\addspace%
            \IfNoValueTF{#5}{%
                \nptextcite[#4]{#6}%
            }{%
                \nptextcite[#4][#5]{#6}%
            }%
        }%
    }
}

\NewDocumentCommand{\tqin}{O{} o m O{} o m m}{
    \textquote{\ignorespaces#7\unskip}%
    \addspace%
    \mkbibparens{%
        \IfNoValueTF{#2}{%
            \nptextcite[#1]{#3}%
        }{%
            \nptextcite[#1][#2]{#3}%
        }%
        \addcomma\addspace\bibstring{quotedin}\addspace%
        \IfNoValueTF{#5}{%
            \nptextcite[#4]{#6}%
        }{%
            \nptextcite[#4][#5]{#6}%
        }%
    }%
}
